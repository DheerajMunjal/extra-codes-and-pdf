        str=""
        for j in i:
            if j=='/':
                break
            else:
                str=str+j
        a.append(str.strip())

a[a=="New Delhi"]="NCR"
a[a=="Gurgaon"]="NCR"
a[a=="Noida"]="NCR"
d={}
for i in a:
    if i=="Bangalore":
        d[i]=d.get(i,0)+1
    if i=="Mumbai":
        d[i]=d.get(i,0)+1
    if i=="NCR":
        d[i]=d.get(i,0)+1
city=[]
nooffundings=[]
for i in d:
    city.append(i)
    nooffundings.append(d[i])
plt.bar(city,nooffundings,width=0.6)

val=""
maxval=-1
